# ====== WINDOWS - Option A : recommandée (mise à jour automatique) ======

# Coller les QUATRE lignes ci-dessous dans PowerShell, dans l'ordre :
irm https://claude.ai/install.ps1 | iex
$bin = "$env:USERPROFILE\.local\bin"; $up = [Environment]::GetEnvironmentVariable("Path", "User")
if (-not $up -or $up -notlike "*$bin*") { [Environment]::SetEnvironmentVariable("Path", ($(if ($up) { "$up;$bin" } else { $bin })), "User") }
$env:Path = [Environment]::GetEnvironmentVariable("Path", "Machine") + ";" + [Environment]::GetEnvironmentVariable("Path", "User")


# ====== WINDOWS - Option B : alternative via winget ======

# Si l'option A ne marche pas. Pensez à lancer 'claude update' une fois par mois.
winget install Anthropic.ClaudeCode

claude
